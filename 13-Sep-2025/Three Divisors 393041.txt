# Problem: Three Divisors - https://leetcode.com/problems/three-divisors/description/?envType=problem-list-v2&envId=number-theory

class Solution {
    public boolean isThree(int n) {
        int count = 0;
        for (int i = 2; i<=n/2; i++) {
            if (n%i == 0) {
                count ++;
            }
        }
        if (count == 1) {
            return true;
        }
        else 
            return false;
    }
}
